158 TreeNode *DPTree::find_best_split(VVD &X_live, vector<double> &gradients_live, int current_depth)
159 {
160  double privacy_budget_for_node;
161  if (params->use_decay) {
162  if (current_depth == 0) {
163  privacy_budget_for_node = tree_params->tree_privacy_budget / (2 * pow(2, params->max_depth + 1) + 2 * pow(2, current_depth + 1));
164  } else {
165  privacy_budget_for_node = tree_params->tree_privacy_budget / (2 * pow(2, current_depth + 1));
166  }
167  } else {
168  privacy_budget_for_node = (tree_params->tree_privacy_budget) / (2 * params->max_depth );
169  }
170 
171  vector<SplitCandidate> probabilities;
172  int lhs_size;
173 
174  // iterate over features
175  for (int feature_index=0; feature_index < dataset->num_x_cols; feature_index++) {
176  bool categorical = std::find((params->cat_idx).begin(), (params->cat_idx).end(), feature_index) != (params->cat_idx).end();
177  std::set<double> unique;
178  for (double feature_value : X_live[feature_index]) {
179  if (std::get<1>(unique.insert(feature_value)) == false){
180  // already had that value
181  continue;
182  }
183  // compute gain
184  double gain = compute_gain(X_live, gradients_live, feature_index, feature_value, lhs_size, categorical);
185  // feature cannot be chosen, skipping
186  if (gain == -1) {
187  continue;
188  }
189  // Gi = epsilon_nleaf * Gi / (2 * delta_G)
190  if(params->use_dp){
191  gain = (privacy_budget_for_node * gain) / (2 * tree_params->delta_g);
192  }
193  SplitCandidate candidate = SplitCandidate(feature_index, feature_value, gain);
194  candidate.lhs_size = lhs_size;
195  candidate.rhs_size = gradients_live.size() - lhs_size;
196  probabilities.push_back(candidate);
197  }
198  }
199 
200  // choose a split using the exponential mechanism
201  int index = exponential_mechanism(probabilities);
202 
203  // construct the node
204  TreeNode *node;
205  if (index == -1) {
206  node = new TreeNode(true);
207  node->left = nullptr;
208  node->right = nullptr;
209  } else {
210  node = new TreeNode(false);
211  node->split_attr = probabilities[index].feature_index;
212  node->split_value = probabilities[index].split_value;
213  node->split_gain = probabilities[index].gain;
214  node->lhs_size = probabilities[index].lhs_size;
215  node->rhs_size = probabilities[index].rhs_size;
216  }
217  node->depth = current_depth;

223  Computes the gain of a split
224 
225  sum(elem : IL)^2 + sum(elem : IR)^2
226  G(IL,IR) = ---------------- ----------------
227  |IL| + lambda |IR| + lambda

229 double DPTree::compute_gain(VVD &samples, vector<double> &gradients_live,
230  int feature_index, double feature_value, int &lhs_size, bool categorical)
231 {
232  // partition into lhs / rhs
233  vector<int> lhs;
234  samples_left_right_partition(lhs, samples, feature_index, feature_value, categorical);
235 
236  int _lhs_size = std::count(lhs.begin(), lhs.end(), 1);
237  int _rhs_size = std::count(lhs.begin(), lhs.end(), 0);
238 
239  lhs_size = _lhs_size;
240 
241  // if all samples go on the same side it's useless to split on this value
242  if ( _lhs_size == 0 or _rhs_size == 0 ) {
243  return -1;
244  }
245 
246  double lhs_gain = 0, rhs_gain = 0;
247  for (size_t index=0; index<lhs.size(); index++) {
248  lhs_gain += lhs[index] * (gradients_live)[index];
249  rhs_gain += (not lhs[index]) * (gradients_live)[index];
250  }
251  lhs_gain = std::pow(lhs_gain,2) / (_lhs_size + params->l2_lambda);
252  rhs_gain = std::pow(rhs_gain,2) / (_rhs_size + params->l2_lambda);
253 
254  double total_gain = lhs_gain + rhs_gain;
255 
256  return std::max(total_gain, 0.0);
257 }

279 // Computes probabilities from the gains. (Larger gain -> larger probability to
280 // be chosen for split). Then a cumulative distribution function is created from
281 // these probabilities. Then we can sample from it using a RNG.
282 // The function returns the index of the chosen split.
283 int DPTree::exponential_mechanism(vector<SplitCandidate> &probs)
284 {
285  // if no split has a positive gain, return. Node will become a leaf
286  int count = std::count_if(probs.begin(), probs.end(),
287  [](SplitCandidate c){ return c.gain > 0; });
288  if (count == 0) {
289  return -1;
290  }
291 
292  // calculate the probabilities from the gains
293  vector<double> gains, probabilities, partials(probs.size());
294  for (auto p : probs) {
295  gains.push_back(p.gain);
296  }
297  double lse = log_sum_exp(gains);
298  for (auto prob : probs) {
299  if (prob.gain <= 0) {
300  probabilities.push_back(0);
301  } else {
302  probabilities.push_back( exp(prob.gain - lse) );
303  }
304  }
305 
306  // non-dp: deterministically choose the best split
307  if (!params->use_dp) {
308  auto max_elem = std::max_element(probabilities.begin(), probabilities.end());
309  // return index of the max_elem
310  return std::distance(probabilities.begin(), max_elem);
311  }
312 
313  // create a cumulative distribution function from the probabilities.
314  // all values will be in [0,1]
315  std::partial_sum(probabilities.begin(), probabilities.end(), partials.begin());
316 
317  double rand01 = ((double) std::rand() / (RAND_MAX)); // [0,1] sollte nicht schiefgehen können TODO TODO
318 
319  // try to find a candidate at least 10 times before giving up and making the node a leaf node
320  for (int tries=0; tries<10; tries++) {
321  for (size_t index=0; index<partials.size(); index++) {
322  if (partials[index] >= rand01) {
323  return index;
324  }
325  }
326  rand01 = ((double) rand() / (RAND_MAX));
327  }
328  return -1;
329 }
